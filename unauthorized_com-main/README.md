# unauthorized_py
# 说明
这是一个自动检测未授权漏洞工具
# 使用
git clone https://github.com/xk11z/unauthorized_com.git 

cd unauthorized_com

apt-get install python3-pip  

pip install -r requirements.txt 

python3 unauthorized_com.py
